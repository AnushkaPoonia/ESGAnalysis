# Summary analysis of Environmental, Social, and Governance (ESG) Scores across different industry groups. It includes summary statistics and visualizations to better understand sectoral trends.
## Load Required Libraries
library(ggplot2)
library(dplyr)
library(tidyr)

# Summary Statistics by Industry Group
industry_group_summary <- csr_data %>%
    group_by(industry_group) %>%
    summarise(
        avg_environment_score = mean(environment_score, na.rm = TRUE),
        avg_social_score = mean(social_score, na.rm = TRUE),
        avg_governance_score = mean(governance_score, na.rm = TRUE),
        avg_total_score = mean(total_score, na.rm = TRUE),
        n = n()  # Number of companies in each industry group
    )

# View the summary table
knitr::kable(industry_group_summary, caption = "Average ESG Scores by Industry Group")

# Boxplots of ESG Scores by Industry Group

# Environmental Scores tend to be higher in industries that focus on sustainability, such as renewable energy and utilities.
# Social Scores may be stronger in industries that prioritize workforce well-being and social responsibility.
# Governance Scores are generally more stable but can fluctuate based on regulatory compliance and leadership structures.
